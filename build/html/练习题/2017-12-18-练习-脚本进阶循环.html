#!/bin/bash
#================================================
#FileName   :guess_46_zhaojiedi.sh
#Author     :zhaojiedi

#Version    :V1.0

declare -i end=32767

#echo $val
#echo $mid
#echo "you val is $val"

while true; do
        read -p "pleater you guesss val[${start}-${end}],default($mid)  " guess
        if [ -z  $guess ] ; then
                guess=$mid
        fi
        if [ "$guess" -eq "$val" ] ; then
                mid=$guess
                break
        elif [ "$guess" -lt "$val" ] ; then
                start=$[guess+1]

        else
                true
        fi

done
